import sqlite3
import pandas as pd
from scipy import stats
import statsmodels.api as sm
import matplotlib.pyplot as plt
import seaborn as sns

# Chemin vers le fichier SQLite
chemin_db = "C:/Users/ARMIDE Informatique/Desktop/Formation pratique/data_immo.sqlite3"

# Connexion  la base de donnes SQLite
conn = sqlite3.connect(chemin_db)

# Cration d'un curseur
cursor = conn.cursor()

try:
    # Exemple de requte SQL pour rcuprer toutes les lignes de la table
    cursor.execute("SELECT valeur_fonciere_actuelle, code_voie FROM bien_immo")

    # Rcupration des rsultats
    rows = cursor.fetchall()
    drows = pd.DataFrame(rows, columns =("valeur_fonciere_actuelle","code_voie"))

    # Affichage des rsultats (pour l'exemple)
    for row in drows:
        print(' ')
        print(drows)

except sqlite3.Error as e:
    print("Erreur lors de l'excution de la requte SQL :", e)

# test de Khi-deux
    # choix des variables 
col1 = 'valeur_fonciere_actuelle'
col2 = 'code_voie'

# Dfinir comme des df
data1 = drows[col1]
data2 = drows[col2]

# effectuer le test 
contingency = pd.crosstab(data1, data2)
chi2_stat, p_value, dof, expected = stats.chi2_contingency(contingency)

# Niveau de signification
alpha = 0.05

# Conclusion du test d'hypothse
if p_value < alpha :
    print('Test de Khi-deux')
    print("Il y a une relation significative entre les deux variables (rejeter l'hypothse nulle)")
else : 
    print("Il n'y a pas suffisamment de preuves pour rejeter l'hypothse nulle")

# Rgression linaire
# Choix des varibales conomiques
x = data1
y = data2

# Ajout de constant  la variable indpendante
x = sm.add_constant(x)

# Rgression linaire
model = sm.OLS(y, x).fit()
predictions = model.predict(x)

# Affichage des rsultats
print(' ')

print('')
print('Rsultats de la rgression linaire')

print(predictions)
